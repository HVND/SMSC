'use strict';
var Request = (function () {
    function Request() {
    }
    Request.prototype.basicAuth = function (userName, userPass) {
        this.userName = userName;
        this.userPass = userPass;
    };
    Request.prototype.httpRequest = function (options) {
        var _this = this;
        return new Promise(function (resolve, reject) {
            var xhr = new XMLHttpRequest();
            xhr.getResponseHeader('Content-Type');
            xhr.open(options.type, options.url, true, options.userName, options.userPass);
            xhr.setRequestHeader('Content-Type', 'application/json');
            xhr.setRequestHeader('Authorization', 'Basic ' +
                btoa(_this.userName + ':' + _this.userPass));
            xhr.onload = function () {
                if (xhr.status === 200) {
                    resolve(xhr.responseText);
                }
                else {
                    reject(new Error(xhr.statusText));
                }
            };
            xhr.send(options.body);
        });
    };
    return Request;
}());
exports.Request = Request;
